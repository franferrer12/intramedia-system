## 4. Docker Compose Completo

**docker-compose.yml:**
```yaml
version: '3.8'

services:
  postgres:
    image: postgres:15-alpine
    container_name: club_postgres
    environment:
      POSTGRES_DB: club_management
      POSTGRES_USER: club_admin
      POSTGRES_PASSWORD: ${DB_PASSWORD}
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./backups:/backups
    ports:
      - "5432:5432"
    networks:
      - club_network
    restart: unless-stopped
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U club_admin"]
      interval: 10s
      timeout: 5s
      retries: 5

  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: club_backend
    environment:
      SPRING_PROFILES_ACTIVE: prod
      DB_HOST: postgres
      DB_PORT: 5432
      DB_NAME: club_management
      DB_USER: club_admin
      DB_PASSWORD: ${DB_PASSWORD}
      JWT_SECRET: ${JWT_SECRET}
      FOURVENUES_API_KEY: ${FOURVENUES_API_KEY}
    ports:
      - "8080:8080"
    depends_on:
      postgres:
        condition: service_healthy
    networks:
      - club_network
    restart: unless-stopped
    volumes:
      - ./uploads:/app/uploads
      - ./logs:/app/logs

  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
      args:
        VITE_API_URL: http://localhost:8080/api
    container_name: club_frontend
    ports:
      - "3000:80"
    depends_on:
      - backend
    networks:
      - club_network
    restart: unless-stopped

  nginx:
    image: nginx:alpine
    container_name: club_nginx
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/ssl:/etc/nginx/ssl:ro
    ports:
      - "80:80"
      - "443:443"
    depends_on:
      - frontend
      - backend
    networks:
      - club_network
    restart: unless-stopped

volumes:
  postgres_data:
    driver: local

networks:
  club_network:
    driver: bridge
```

**backend/Dockerfile:**
```dockerfile
FROM maven:3.9-eclipse-temurin-17 AS build
WORKDIR /app
COPY pom.xml .
RUN mvn dependency:go-offline
COPY src ./src
RUN mvn clean package -DskipTests

FROM eclipse-temurin:17-jre-alpine
WORKDIR /app
COPY --from=build /app/target/club-management-*.jar app.jar
EXPOSE 8080
ENTRYPOINT ["java", "-jar", "app.jar"]
```

**frontend/Dockerfile:**
```dockerfile
FROM node:18-alpine AS build
WORKDIR /app
COPY package*.json ./
RUN npm ci
COPY . .
ARG VITE_API_URL
ENV VITE_API_URL=$VITE_API_URL
RUN npm run build

FROM nginx:alpine
COPY --from=build /app/dist /usr/share/nginx/html
COPY nginx.conf /etc/nginx/conf.d/default.conf
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
```

**frontend/nginx.conf:**
```nginx
server {
    listen 80;
    server_name _;
    root /usr/share/nginx/html;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }

    location /api {
        proxy_pass http://backend:8080;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    gzip on;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript;
}
```

### scripts/backup.sh
```bash
#!/bin/bash

# Configuraci√≥n
BACKUP_DIR="./backups"
DB_CONTAINER="club_postgres"
DB_NAME="club_management"
DB_USER="club_admin"
RETENTION_DAYS=30
DATE=$(date +%Y%m%d_%H%M%S)

# Colores
GREEN='\033[0;32m'
RED='\033[0;31m'
NC='\033[0m'

# Crear directorio si no existe
mkdir -p $BACKUP_DIR

echo "üíæ Iniciando backup de PostgreSQL..."

# Backup de la base de datos
if docker exec $DB_CONTAINER pg_dump -U $DB_USER $DB_NAME | gzip > $BACKUP_DIR/backup_$DATE.sql.gz; then
    echo -e "${GREEN}‚úÖ Backup completado: backup_$DATE.sql.gz${NC}"
    
    # Tama√±o del backup
    SIZE=$(du -h $BACKUP_DIR/backup_$DATE.sql.gz | cut -f1)
    echo "üì¶ Tama√±o: $SIZE"
else
    echo -e "${RED}‚ùå ERROR: Backup fall√≥${NC}"
    exit 1
fi

# Eliminar backups antiguos
echo "üßπ Limpiando backups antiguos (>$RETENTION_DAYS d√≠as)..."
find $BACKUP_DIR -name "backup_*.sql.gz" -mtime +$RETENTION_DAYS -delete
echo "‚úÖ Limpieza completada"

# Listar backups
echo ""
echo "üìã Backups disponibles:"
ls -lh $BACKUP_DIR/backup_*.sql.gz 2>/dev/null || echo "No hay backups"

# Opcional: Subir a Google Drive con rclone
# if command -v rclone &> /dev/null; then
#     echo "‚òÅÔ∏è  Subiendo a Google Drive..."
#     rclone copy $BACKUP_DIR/backup_$DATE.sql.gz gdrive:club-backups/
# fi

echo ""
echo "‚ú® Proceso de backup finalizado"
```
**Uso:**
```bash

# Programar backup diario (crontab)

```

### scripts/restore.sh
```bash
#!/bin/bash

# Verificar argumento
if [ -z "$1" ]; then
    echo "‚ùå Error: Debe especificar el archivo de backup"
    echo "Uso: ./restore.sh /path/to/backup_YYYYMMDD_HHMMSS.sql.gz"
    echo ""
    echo "üìã Backups disponibles:"
    ls -lh backups/backup_*.sql.gz 2>/dev/null || echo "No hay backups"
    exit 1
fi

BACKUP_FILE=$1
DB_CONTAINER="club_postgres"
DB_NAME="club_management"
DB_USER="club_admin"

# Verificar que el archivo existe
if [ ! -f "$BACKUP_FILE" ]; then
    echo "‚ùå Error: Archivo no encontrado: $BACKUP_FILE"
    exit 1
fi

echo "‚ö†Ô∏è  ADVERTENCIA: Esto SOBREESCRIBIR√Å la base de datos actual."
echo "üìÅ Archivo: $BACKUP_FILE"
read -p "¬øEst√°s seguro? Escribe 'SI' para confirmar: " confirm

if [ "$confirm" != "SI" ]; then
    echo "‚ùå Restauraci√≥n cancelada"
    exit 0
fi

echo "üíæ Creando backup de seguridad antes de restaurar..."
./scripts/backup.sh

echo "üîÑ Restaurando base de datos..."

# Descomprimir y restaurar
if gunzip -c $BACKUP_FILE | docker exec -i $DB_CONTAINER psql -U $DB_USER -d $DB_NAME; then
    echo "‚úÖ Restauraci√≥n completada exitosamente"
else
    echo "‚ùå ERROR: Restauraci√≥n fall√≥"
    exit 1
fi

echo "üîÑ Reiniciando servicios..."
docker-compose restart backend

echo "‚ú® Proceso completado"
```

## 8. Troubleshooting

### Backend no inicia

**Problema:** `Port 8080 already in use`
```bash
# Encontrar proceso usando el puerto
lsof -i :8080
# o
netstat -tuln | grep 8080

# Matar proceso
kill -9 <PID>
```

**Problema:** `Could not connect to database`
```bash
# Verificar que PostgreSQL est√° corriendo
docker ps | grep postgres

# Ver logs de PostgreSQL
docker logs club_postgres

# Verificar credenciales en .env
cat .env | grep DB_
```

**Problema:** `Flyway migration failed`
```bash
# Ver estado de migraciones
docker exec club_postgres psql -U club_admin -d club_management -c "SELECT * FROM flyway_schema_history;"

# Reparar migraciones (¬°cuidado!)
# Solo si sabes lo que haces
docker exec -it club_backend ./mvnw flyway:repair
```

### Frontend no carga

**Problema:** `Cannot connect to backend`
```bash
# Verificar API URL
cat frontend/.env | grep VITE_API_URL

# Probar endpoint directamente
curl http://localhost:8080/api/eventos

# Verificar CORS
# En DevTools ‚Üí Network ‚Üí Ver headers de respuesta
```

**Problema:** `npm install fails`
```bash
# Limpiar cache
npm cache clean --force
rm -rf node_modules package-lock.json
npm install
```

### Docker issues

**Problema:** Servicios no levantan
```bash
# Ver logs completos
docker-compose logs

# Rebuild desde cero
docker-compose down -v
docker-compose build --no-cache
docker-compose up -d

# Ver recursos
docker stats
```

**Problema:** Out of disk space
